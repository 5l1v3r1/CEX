367  // Different computer architectures store data using different byte orders. "Big-endian"
368  // means the most significant byte is on the left end of a word. "Little-endian" means the
369  // most significant byte is on the right end of a word. i.e.:
370  // BE: uint(block[3]) | (uint(block[2]) << 8) | (uint(block[1]) << 16) | (uint(block[0]) << 24)
371  // LE: uint(block[0]) | (uint(block[1]) << 8) | (uint(block[2]) << 16) | (uint(block[3]) << 24)

394  template<typename ArrayA, typename ArrayB>
395  inline static void BlockToBe(const ArrayA &Input, size_t InOffset, ArrayB &Output, size_t OutOffset, size_t Length)
396  {
397  CexAssert(sizeof(ArrayA::value_type) == sizeof(byte), "Input must be a byte array");
398  CexAssert((Input.size() - InOffset) >= Length, "Length is larger than input capacity");
399  CexAssert((Output.size() - OutOffset) * sizeof(ArrayB::value_type) >= Length, "Length is larger than output capacity");
400 
401 #if defined(IS_BIG_ENDIAN)
402  Utility::MemUtils::Copy(Input, InOffset, Output, OutOffset, Length);
403 #else
404  const size_t VARSZE = sizeof(ArrayB::value_type);
405 
406  for (size_t i = 0; i < Length / VARSZE; ++i)
407  {
408  for (size_t j = VARSZE; j > 0; --j)
409  {
410  Output[OutOffset + i] |= static_cast<ArrayB::value_type>(Input[InOffset + (i * VARSZE) + (VARSZE - j)]) << (8 * (j - 1));
411  }
412  }
413 #endif
414  }

683  template<typename Array>
684  inline static void BeIncrease8(const Array &Input, Array &Output, const size_t Length)
685  {
686  CexAssert(sizeof(Array::value_type) == sizeof(byte), "Input and Output must be an array of 8bit integers");
687  CexAssert(!std::is_signed<Array::value_type>::value, "Input and Output must be an unsigned integer array");
688 
689  const int CTRSZE = static_cast<int>(Output.size() - 1);
690  uint ctrLen = static_cast<uint>(Length);
691  std::array<byte, sizeof(uint)> ctrInc;
692 
693  std::memcpy(&ctrInc[0], &ctrLen, ctrInc.size());
694  std::memcpy(&Output[0], &Input[0], Input.size());
695  byte carry = 0;
696 
697  for (int i = CTRSZE; i >= 0; --i)
698  {
699  byte odst = Output[i];
700  byte osrc = CTRSZE - i < static_cast<int>(ctrInc.size()) ? static_cast<byte>(ctrInc[CTRSZE - i]) : static_cast<byte>(0);
701  byte ndst = static_cast<byte>(odst + osrc + carry);
702  carry = ndst < odst ? 1 : 0;
703  Output[i] = ndst;
704  }
705  }
