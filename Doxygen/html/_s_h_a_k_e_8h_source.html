270  template<typename Array>
271  static size_t LeftEncode(Array &Buffer, size_t Offset, size_t Value)
272  {
273  size_t i;
274  size_t n;
275  size_t v;
276 
277  for (v = Value, n = 0; v && (n < sizeof(size_t)); ++n, v >>= 8);
278 
279  if (n == 0)
280  {
281  n = 1;
282  }
283 
284  for (i = 1; i <= n; ++i)
285  {
286  Buffer[Offset + i] = (uint8_t)(Value >> (8 * (n - i)));
287  }
288 
289  Buffer[Offset] = (uint8_t)n;
290 
291  return (n + 1);
292  }
