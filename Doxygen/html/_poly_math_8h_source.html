84  inline static void Add(Array &R, const Array &A, const Array &B, int Q)
85  {
86  const T VN(5);
87 
88 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
89  const size_t ULVLEN = T::size() / sizeof(uint);
90  std::array<uint, ULVLEN> tmpR;
91  const T NQ(Q);
92  T tmpA, tmpB;
93 #else
94  const size_t ULVLEN = 1;
95 #endif
96 
97  for (size_t i = 0; i < R.size(); i += ULVLEN)
98  {
99 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
100 # if defined(__AVX512__)
101  tmpA.Load(A[i + 15], A[i + 14], A[i + 13], A[i + 12], A[i + 11], A[i + 10], A[i + 9], A[i + 8], A[i + 7], A[i + 6], A[i + 5], A[i + 4], A[i + 3], A[i + 2], A[i + 1], A[i]);
102  tmpB.Load(B[i + 15], B[i + 14], B[i + 13], B[i + 12], B[i + 11], B[i + 10], B[i + 9], B[i + 8], B[i + 7], B[i + 6], B[i + 5], B[i + 4], B[i + 3], B[i + 2], B[i + 1], B[i]);
103 # elif defined(__AVX2__)
104  tmpA.Load(A[i + 7], A[i + 6], A[i + 5], A[i + 4], A[i + 3], A[i + 2], A[i + 1], A[i]);
105  tmpB.Load(B[i + 7], B[i + 6], B[i + 5], B[i + 4], B[i + 3], B[i + 2], B[i + 1], B[i]);
106 # elif defined(__AVX__)
107  tmpA.Load(A[i + 3], A[i + 2], A[i + 1], A[i]);
108  tmpB.Load(B[i + 3], B[i + 2], B[i + 1], B[i]);
109 # endif
110 
111  T VF(tmpA + tmpB);
112  T VU = (VF * VN) >> 16;
113  VU *= NQ;
114  VF -= VU;
115  VF.Store(tmpR, 0);
116 
117  for (size_t j = 0; j < ULVLEN; ++j)
118  {
119  R[j + i] = static_cast<ushort>(tmpR[j]);
120  }
121 #else
122  T F = A[i] + B[i];
123  uint U = static_cast<uint>(F * VN) >> 16;
124  U *= Q;
125  F -= U;
126  R[i] = F;
127 #endif
128  }

149  template <typename Array, class T>
150  inline static void Mul(Array &R, const Array &Factors, int Q, uint QInv, uint RLog)
151  {
152 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
153  const size_t ULVLEN = T::size() / sizeof(uint);
154  std::array<uint, ULVLEN> tmpR;
155  T tmpP, tmpF;
156 #else
157  const size_t ULVLEN = 1;
158 #endif
159 
160  for (size_t i = 0; i < R.size(); i += ULVLEN)
161  {
162 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
163 # if defined(__AVX512__)
164  tmpP.Load(R[i + 15], R[i + 14], R[i + 13], R[i + 12], R[i + 11], R[i + 10], R[i + 9], R[i + 8], R[i + 7], R[i + 6], R[i + 5], R[i + 4], R[i + 3], R[i + 2], R[i + 1], R[i]);
165  tmpF.Load(Factors[i + 15], Factors[i + 14], Factors[i + 13], Factors[i + 12], Factors[i + 11], Factors[i + 10], Factors[i + 9], Factors[i + 8], Factors[i + 7], Factors[i + 6], Factors[i + 5], Factors[i + 4], Factors[i + 3], Factors[i + 2], Factors[i + 1], Factors[i]);
166 # elif defined(__AVX2__)
167  tmpP.Load(R[i + 7], R[i + 6], R[i + 5], R[i + 4], R[i + 3], R[i + 2], R[i + 1], R[i]);
168  tmpF.Load(Factors[i + 7], Factors[i + 6], Factors[i + 5], Factors[i + 4], Factors[i + 3], Factors[i + 2], Factors[i + 1], Factors[i]);
169 # elif defined(__AVX__)
170  tmpP.Load(R[i + 3], R[i + 2], R[i + 1], R[i]);
171  tmpF.Load(Factors[i + 3], Factors[i + 2], Factors[i + 1], Factors[i]);
172 # endif
173 
174  T a = tmpP * tmpF;
175  T u = (a * T(QInv));
176  u &= ((T::ONE() << RLog) - T::ONE());
177  u *= T(Q);
178  a += u;
179  a >>= 18;
180  a.Store(tmpR, 0);
181 
182  for (size_t j = 0; j < ULVLEN; ++j)
183  {
184  R[j + i] = static_cast<ushort>(tmpR[j]);
185  }
186 #else
187  T a = R[i] * Factors[i];
188  T u = (a * QInv);
189  u &= ((1 << RLog) - 1);
190  u *= Q;
191  a += u;
192  R[i] = a >> 18;
193 #endif
194  }
195  }
