31 static void THXDecryptW(const std::vector<byte> &Input, const size_t InOffset, std::vector<byte> &Output, const size_t OutOffset, std::vector<uint> &Key, std::vector<uint> &Sbox)
32 {
33 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
34 
35  const size_t RNDCNT = 8;
36  const size_t INPOFF = T::size();
37  size_t keyCtr = 4;
38 
39  // input round
40  T X2(Input, InOffset);
41  T X3(Input, InOffset + INPOFF);
42  T X0(Input, InOffset + (INPOFF * 2));
43  T X1(Input, InOffset + (INPOFF * 3));
44  T::Transpose(X2, X3, X0, X1);
45  T T0, T1;
46  T N2(2);
47 
48  X2 ^= T(Key[keyCtr]);
49  ++keyCtr;
50  X3 ^= T(Key[keyCtr]);
51  ++keyCtr;
52  X0 ^= T(Key[keyCtr]);
53  ++keyCtr;
54  X1 ^= T(Key[keyCtr]);
55  keyCtr = Key.size();
56 
57  do
58  {
59  T0 = Fe0W(X2, Sbox);
60  T1 = Fe3W(X3, Sbox);
61  --keyCtr;
62  X1 ^= T0 + N2 * T1 + T(Key[keyCtr]);
63  X0 = (X0 << 1) | (X0 >> 31);
64  --keyCtr;
65  X0 ^= (T0 + T1 + T(Key[keyCtr]));
66  X1 = (X1 >> 1) | (X1 << 31);
67 
68  T0 = Fe0W(X0, Sbox);
69  T1 = Fe3W(X1, Sbox);
70  --keyCtr;
71  X3 ^= T0 + N2 * T1 + T(Key[keyCtr]);
72  X2 = (X2 << 1) | (X2 >> 31);
73  --keyCtr;
74  X2 ^= (T0 + T1 + T(Key[keyCtr]));
75  X3 = (X3 >> 1) | (X3 << 31);
76  }
77  while (keyCtr != RNDCNT);
78 
79  // last round
80  keyCtr = 0;
81  X0 ^= T(Key[keyCtr]);
82  ++keyCtr;
83  X1 ^= T(Key[keyCtr]);
84  ++keyCtr;
85  X2 ^= T(Key[keyCtr]);
86  ++keyCtr;
87  X3 ^= T(Key[keyCtr]);
88 
89  T::Transpose(X0, X1, X2, X3);
90  X0.Store(Output, OutOffset);
91  X1.Store(Output, OutOffset + INPOFF);
92  X2.Store(Output, OutOffset + (INPOFF * 2));
93  X3.Store(Output, OutOffset + (INPOFF * 3));
94 
95 #endif
96 }

104  const size_t INPOFF = T::size();
105  size_t keyCtr = 0;
106 
107  // input round
108  T X0(Input, InOffset);
109  T X1(Input, InOffset + INPOFF);
110  T X2(Input, InOffset + (INPOFF * 2));
111  T X3(Input, InOffset + (INPOFF * 3));
112  T::Transpose(X0, X1, X2, X3);
113  T T0, T1;
114  T N2(2);
115 
116  X0 ^= T(Key[keyCtr]);
117  ++keyCtr;
118  X1 ^= T(Key[keyCtr]);
119  ++keyCtr;
120  X2 ^= T(Key[keyCtr]);
121  ++keyCtr;
122  X3 ^= T(Key[keyCtr]);
123  keyCtr = 7;
124 
125  do
126  {
127  T0 = Fe0W(X0, Sbox);
128  T1 = Fe3W(X1, Sbox);
129  ++keyCtr;
130  X2 ^= T0 + T1 + T(Key[keyCtr]);
131  X2 = (X2 >> 1) | (X2 << 31);
132  X3 = (X3 << 1) | (X3 >> 31);
133  ++keyCtr;
134  X3 ^= (T0 + N2 * T1 + T(Key[keyCtr]));
135 
136  T0 = Fe0W(X2, Sbox);
137  T1 = Fe3W(X3, Sbox);
138  ++keyCtr;
139  X0 ^= T0 + T1 + T(Key[keyCtr]);
140  X0 = (X0 >> 1) | (X0 << 31);
141  X1 = ((X1 << 1) | (X1 >> 31));
142  ++keyCtr;
143  X1 ^= (T0 + N2 * T1 + T(Key[keyCtr]));
144  }
145  while (keyCtr != RNDCNT);
146 
147  // last round
148  keyCtr = 4;
149  X2 ^= T(Key[keyCtr]);
150  ++keyCtr;
151  X3 ^= T(Key[keyCtr]);
152  ++keyCtr;
153  X0 ^= T(Key[keyCtr]);
154  ++keyCtr;
155  X1 ^= T(Key[keyCtr]);
156 
157  T::Transpose(X2, X3, X0, X1);
158  X2.Store(Output, OutOffset);
159  X3.Store(Output, OutOffset + INPOFF);
160  X0.Store(Output, OutOffset + (INPOFF * 2));
161  X1.Store(Output, OutOffset + (INPOFF * 3));
162 
163 #endif
