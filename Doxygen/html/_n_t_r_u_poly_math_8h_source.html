59  inline static int32_t I16NegativeMask(int16_t X)
60  {
61  // return -1 if x<0; otherwise return 0
62  ushort u;
63 
64  u = X;
65  u >>= 15;
66 
67  // alternative with gcc -fwrapv:
68  // x>>15 compiles to CPU's arithmetic right shift
69  return -static_cast<int32_t>(u);
70  }

72  inline static int32_t I16NonZeroMask(int16_t X)
73  {
74  // return -1 if x!=0; else return 0
75  ushort u;
76  uint v;
77 
78  u = X;
79  v = u;
80  // 0, else 2^32-65535...2^32-1
81  v = ~v + 1;
82  // 0, else 1
83  v >>= 31;
84 
85  // 0, else -1

99  inline static void I32DivModU14(int32_t &Q, ushort &R, int32_t X, ushort M)
100  {
101  uint mask;
102  uint uq;
103  uint uq2;
104  ushort ur;
105  ushort ur2;
106 
107  U32DivModU14(uq, ur, 0x80000000UL + static_cast<uint>(X), M);
108  U32DivModU14(uq2, ur2, 0x80000000UL, M);
109  ur -= ur2;
110  uq -= uq2;
111  mask = ~static_cast<uint>(ur >> 15) + 1;
112  ur += mask & M;
113  uq += mask;
114  R = ur;
115  Q = uq;
116  }

130  template <typename ArrayU32>
131  static void MinMax(ArrayU32 &X, size_t XOffset, ArrayU32 &Y, size_t YOffset)
132  {
133  uint xi;
134  uint yi;
135  uint xy;
136  uint c;
137 
138  xi = X[XOffset];
139  yi = Y[YOffset];
140  xy = xi ^ yi;
141  c = yi - xi;
142 
143  c ^= xy & (c ^ yi ^ 0x80000000UL);
144  c >>= 31;
145  c = ~c + 1;
146  c &= xy;
147  X[XOffset] = xi ^ c;
148  Y[YOffset] = yi ^ c;
149  }

151  template <typename ArrayU32>
152  static void U32Sort(ArrayU32 &X, int32_t N)
153  {
154  size_t i;
155  size_t pt;
156  size_t qt;
157  size_t top;
158 
159  if (N < 2)
160  {
161  return;
162  }
163 
164  top = 1;
165 
166  while (top < N - top)
167  {
168  top += top;
169  }
170 
171  for (pt = top; pt > 0; pt >>= 1)
172  {
173  for (i = 0; i < N - pt; ++i)
174  {
175  if (!(i & pt))
176  {
177  MinMax(X, i, X, i + pt);
178  }
179  }
180 
181  for (qt = top; qt > pt; qt >>= 1)
182  {
183  for (i = 0; i < N - qt; ++i)
184  {
185  if (!(i & pt))
186  {
187  MinMax(X, i + pt, X, i + qt);
188  }
189  }
190  }
191  }
192  }
