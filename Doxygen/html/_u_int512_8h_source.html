412  static void Transpose(UInt512 &X0, UInt512 &X1, UInt512 &X2, UInt512 &X3)
413  {
414  __m512i T0 = _mm512_unpacklo_epi32(X0.zmm, X1.zmm);
415  __m512i T1 = _mm512_unpacklo_epi32(X2.zmm, X3.zmm);
416  __m512i T2 = _mm512_unpackhi_epi32(X0.zmm, X1.zmm);
417  __m512i T3 = _mm512_unpackhi_epi32(X2.zmm, X3.zmm);
418  X0.zmm = _mm512_unpacklo_epi64(T0, T1);
419  X1.zmm = _mm512_unpackhi_epi64(T0, T1);
420  X2.zmm = _mm512_unpacklo_epi64(T2, T3);
421  X3.zmm = _mm512_unpackhi_epi64(T2, T3);
422  }
