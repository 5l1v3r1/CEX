240  template<typename Array>
241  static size_t LeftEncode(Array &Buffer, size_t Offset, size_t Value)
242  {
243  size_t i;
244  size_t n;
245  size_t v;
246 
247  for (v = Value, n = 0; v && (n < sizeof(size_t)); ++n, v >>= 8);
248 
249  if (n == 0)
250  {
251  n = 1;
252  }
253 
254  for (i = 1; i <= n; ++i)
255  {
256  Buffer[Offset + i] = (uint8_t)(Value >> (8 * (n - i)));
257  }
258 
259  Buffer[Offset] = (uint8_t)n;
260 
261  return (n + 1);
262  }
