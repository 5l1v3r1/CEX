768  inline static void BeIncrease8(Array &Output, const size_t Length)
769  {
770  CEXASSERT(sizeof(Array::value_type) == sizeof(byte), "Input and Output must be an array of 8bit integers");

772 
773  const int CTRLEN = static_cast<int>(Output.size() - 1);
774  uint ctrLen = static_cast<uint>(Length);
775  std::array<byte, sizeof(uint)> ctrInc;
776 
777  std::memcpy(&ctrInc[0], &ctrLen, ctrInc.size());
778  byte carry = 0;
779 
780  for (int i = CTRLEN; i >= 0; --i)
781  {
782  byte odst = Output[i];

784  byte ndst = static_cast<byte>(odst + osrc + carry);
785  carry = ndst < odst ? 1 : 0;
