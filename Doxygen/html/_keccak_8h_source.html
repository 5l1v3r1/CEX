118  template<typename Array>
119  static void Absorb(const Array &Input, size_t InOffset, size_t InLength, std::array<ulong, KECCAK_STATE_SIZE> &State)
120  {
121  CEXASSERT(InLength % sizeof(ulong) == 0, "The input length is not 64-bit aligned");
122 
123 #if defined(CEX_IS_LITTLE_ENDIAN)
124  MemoryTools::XOR(Input, InOffset, State, 0, InLength);
125 #else
126  for (size_t i = 0; i < InLength / sizeof(ulong); ++i)
127  {
128  State[i] ^= IntegerTools::LeBytesTo64(Input, InOffset + (i * sizeof(ulong)));
129  }
130 #endif
131  }

143  static void SqueezeR24(std::array<ulong, KECCAK_STATE_SIZE> &State, Array &Output, size_t OutOffset, size_t Blocks, size_t Rate)
144  {
145  while (Blocks > 0)
146  {
147 #if defined(CEX_DIGEST_COMPACT)
148  Digest::Keccak::PermuteR24P1600C(State);
149 #else
150  Digest::Keccak::PermuteR24P1600U(State);
151 #endif
152 
153 #if defined(CEX_IS_LITTLE_ENDIAN)
154  MemoryTools::Copy(State, 0, Output, OutOffset, Rate);
155 #else
156 
157  for (size_t i = 0; i < (Rate >> 3); i++)
158  {
159  IntegerTools::Le64ToBytes(State[i], Output, OutOffset + (8 * i));
160  }
161 #endif
162 
163  OutOffset += Rate;
164  --Blocks;
165  }

215  template<typename ArrayA, typename ArrayB>
216  static void XOFR24P1600(const ArrayA &Input, size_t InOffset, size_t InLength, ArrayB &Output, size_t OutOffset, size_t OutLength, size_t Rate)
217  {
218  std::array<byte, 200> msg = { 0 };
219  std::array<ulong, 25> state = { 0 };
220  size_t blkcnt;
221  size_t i;
222 
223  while (InLength >= Rate)
224  {
225  Keccak::Absorb(Input, InOffset, Rate, state);
226 #if defined(CEX_DIGEST_COMPACT)
227  Keccak::PermuteR24P1600C(state);
228 #else
229  Keccak::PermuteR24P1600U(state);
230 #endif
231  InLength -= Rate;
232  InOffset += Rate;
233  }
234 
235  MemoryTools::Copy(Input, InOffset, msg, 0, InLength);
236  msg[InLength] = 0x1F;
237  MemoryTools::Clear(msg, InLength + 1, Rate - InLength + 1);
238  msg[Rate - 1] |= 128;
239 
240 #if defined(CEX_IS_LITTLE_ENDIAN)
241  MemoryTools::XOR(msg, 0, state, 0, Rate);
242 #else
243  for (i = 0; i < (Rate >> 3); ++i)
244  {
245  state[i] ^= IntegerTools::LeBytesTo64(msg, (8 * i));
246  }
247 #endif
248 
249  blkcnt = OutLength / Rate;
250  Keccak::SqueezeR24(state, Output, OutOffset, blkcnt, Rate);
251  OutOffset += blkcnt * Rate;
252  OutLength -= blkcnt * Rate;
253 
254  if (OutLength != 0)
255  {
256 #if defined(CEX_DIGEST_COMPACT)
257  Digest::Keccak::PermuteR24P1600C(state);
258 #else
259  Keccak::PermuteR24P1600U(state);
260 #endif
261 
262  const size_t FNLBLK = OutLength % sizeof(ulong) == 0 ? OutLength / sizeof(ulong) : OutLength / sizeof(ulong) + 1;
263 
264  for (i = 0; i < FNLBLK; i++)
265  {
266  IntegerTools::Le64ToBytes(state[i], msg, (8 * i));
267  }
268 
269  MemoryTools::Copy(msg, 0, Output, OutOffset, OutLength);
270  }
271  }
