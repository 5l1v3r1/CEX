184  template<typename ArrayU8>
185  static void AbsorbR24(const ArrayU8 &Input, size_t InOffset, size_t InLength, size_t Rate, byte Domain, std::array<ulong, KECCAK_STATE_SIZE> &State)
186  {
187  std::array<byte, 200> msg = { 0 };
188 
189  while (InLength >= Rate)
190  {
191  Keccak::FastAbsorb(Input, InOffset, Rate, State);
192 
193 #if defined(CEX_DIGEST_COMPACT)
194  Keccak::PermuteR24P1600C(State);
195 #else
196  Keccak::PermuteR24P1600U(State);
197 #endif
198 
199  InLength -= Rate;
200  InOffset += Rate;
201  }
202 
203  MemoryTools::Copy(Input, InOffset, msg, 0, InLength);
204  msg[InLength] = Domain;
205  msg[Rate - 1] |= 128;
206 
207 #if defined(CEX_IS_LITTLE_ENDIAN)
208  MemoryTools::XOR(msg, 0, State, 0, Rate);
209 #else
210  for (i = 0; i < (Rate >> 3); ++i)
211  {
212  State[i] ^= IntegerTools::LeBytesTo64(msg, (8 * i));
213  }
214 #endif
215  }

297  template<typename ArrayU8>
298  static void CustomizeR24(const ArrayU8 &Customization, const ArrayU8 &Name, size_t Rate, std::array<ulong, KECCAK_STATE_SIZE> &State)
299  {
300  const size_t BUFFER_SIZE = KECCAK_STATE_SIZE * sizeof(ulong);
301  std::array<byte, BUFFER_SIZE> pad = { 0 };
302  size_t i;
303  size_t offset;
304 
305  offset = Keccak::LeftEncode(pad, 0, static_cast<ulong>(Rate));
306  offset += Keccak::LeftEncode(pad, offset, static_cast<ulong>(Name.size()) * 8);
307 
308  if (Name.size() != 0)
309  {
310  for (i = 0; i < Name.size(); ++i)
311  {
312  if (offset == Rate)
313  {
314  Keccak::FastAbsorb(pad, 0, Rate, State);
315  PermuteR24P1600U(State);
316  offset = 0;
317  }
318 
319  pad[offset] = Name[i];
320  ++offset;
321  }
322  }
323 
324  offset += Keccak::LeftEncode(pad, offset, static_cast<ulong>(Customization.size()) * 8);
325 
326  if (Customization.size() != 0)
327  {
328  for (i = 0; i < Customization.size(); ++i)
329  {
330  if (offset == Rate)
331  {
332  Keccak::FastAbsorb(pad, 0, Rate, State);
333  PermuteR24P1600U(State);
334  offset = 0;
335  }
336 
337  pad[offset] = Customization[i];
338  ++offset;
339  }
340  }
341 
342  MemoryTools::Clear(pad, offset, BUFFER_SIZE - offset);
343  offset = (offset % sizeof(ulong) == 0) ? offset : offset + (sizeof(ulong) - (offset % sizeof(ulong)));
344  MemoryTools::XOR(pad, 0, State, 0, offset);
345 
346  PermuteR24P1600U(State);
347  }

479  static void FastAbsorb(const ArrayU8 &Input, size_t InOffset, size_t InLength, std::array<ulong, KECCAK_STATE_SIZE> &State)
480  {
481  CEXASSERT(InLength % sizeof(ulong) == 0, "The input length is not 64-bit aligned");
482 
483 #if defined(CEX_IS_LITTLE_ENDIAN)
484  MemoryTools::XOR(Input, InOffset, State, 0, InLength);
485 #else
486  for (size_t i = 0; i < InLength / sizeof(ulong); ++i)
487  {
488  State[i] ^= IntegerTools::LeBytesTo64(Input, InOffset + (i * sizeof(ulong)));
489  }
490 #endif

502  template<typename ArrayU8>
503  static ulong LeftEncode(ArrayU8 &Output, size_t Offset, ulong Value)
504  {
505  ulong i;
506  ulong n;
507  ulong v;
508 
509  for (v = Value, n = 0; v && (n < sizeof(ulong)); ++n, v >>= 8)
510  {
511  }
512 
513  if (n == 0)
514  {
515  n = 1;
516  }
517 
518  for (i = 1; i <= n; ++i)
519  {
520  Output[Offset + i] = static_cast<uint8_t>(Value >> (8 * (n - i)));
521  }
522 
523  Output[Offset] = static_cast<uint8_t>(n);
524 
525  return (n + 1);
526  }

540  template<typename ArrayU8>
541  static void MACR24P1600(const ArrayU8 &Key, const ArrayU8 &Customization, const ArrayU8 &Message, size_t Offset, size_t Length, ArrayU8 &Output, size_t Rate)
542  {
543  const size_t BUFLEN = Keccak::KECCAK_STATE_SIZE * sizeof(ulong);
544  const std::vector<byte> KNAME{ 0x4B, 0x4D, 0x41, 0x43 };
545  std::array<ulong, KECCAK_STATE_SIZE> state = { 0 };
546  std::array<byte, BUFLEN> pad = { 0 };
547  std::vector<byte> buf(sizeof(size_t) + 1);
548  ulong poft;
549  size_t blen;
550  size_t i;
551 
552  // add the customization
553  CustomizeR24(Customization, KNAME, Rate, state);
554 
555  // absorb the key
556  poft = LeftEncode(pad, 0, static_cast<ulong>(Rate));
557  poft += LeftEncode(pad, poft, static_cast<ulong>(Key.size()) * sizeof(ulong));
558 
559  if (Key.size() != 0)
560  {
561  for (i = 0; i < Key.size(); ++i)
562  {
563  if (poft == Rate)
564  {
565  FastAbsorb(pad, 0, Rate, state);
566  PermuteR24P1600U(state);
567  poft = 0;
568  }
569 
570  pad[poft] = Key[i];
571  ++poft;
572  }
573  }
574 
575  MemoryTools::Clear(pad, poft, BUFLEN - poft);
576  poft = (poft % sizeof(ulong) == 0) ? poft : poft + (sizeof(ulong) - (poft % sizeof(ulong)));
577 
578  for (i = 0; i < poft; i += 8)
579  {
580  state[i / sizeof(ulong)] ^= IntegerTools::LeBytesTo64(pad, i);
581  }
582 
583  PermuteR24P1600U(state);
584 
585  // loop through the message, absorbing it into the state
586  while (Length >= Rate)
587  {
588  FastAbsorb(Message, Offset, Rate, state);
589  PermuteR24P1600U(state);
590  Length -= Rate;
591  Offset += Rate;
592  }
593 
594  if (Length != 0)
595  {
596  MemoryTools::Copy(Message, Offset, pad, 0, Length);
597  MemoryTools::Clear(pad, Length, pad.size() - Length);
598  }
599 
600  // add the remaining message bytes to the state and finalize
601 
602  blen = Keccak::RightEncode(pad, Length, static_cast<ulong>(Output.size()) * sizeof(ulong));
603 
604  Length += blen;
605  pad[Length] = Keccak::KECCAK_KMAC_DOMAIN;
606  pad[Rate - 1] |= 128;
607  Keccak::FastAbsorb(pad, 0, Rate, state);
608 
609  // output the hash
610  Length = Output.size();
611  Offset = 0;
612 
613  while (Length != 0)
614  {
615  const size_t DIFF = IntegerTools::Min(Rate, Length);
616  PermuteR24P1600U(state);
617  MemoryTools::Copy(state, 0, Output, Offset, DIFF);
618  Offset += DIFF;
619  Length -= DIFF;
620  }
621  }

793  static void XOFR24P1600(const ArrayU8A &Input, size_t InOffset, size_t InLength, ArrayU8B &Output, size_t OutOffset, size_t OutLength, size_t Rate)
794  {
795  std::array<byte, KECCAK_STATE_SIZE * sizeof(ulong)> msg = { 0 };
796  std::array<ulong, KECCAK_STATE_SIZE> state = { 0 };
797  size_t blkcnt;
798  size_t i;
799 
800  while (InLength >= Rate)
801  {
802  FastAbsorb(Input, InOffset, Rate, state);
803  PermuteR24P1600U(state);
804  InLength -= Rate;
805  InOffset += Rate;
806  }
807 
808  MemoryTools::Copy(Input, InOffset, msg, 0, InLength);
809  msg[InLength] = KECCAK_SHAKE_DOMAIN;
810  MemoryTools::Clear(msg, InLength + 1, Rate - InLength + 1);
811  msg[Rate - 1] |= 128;
812 
813 #if defined(CEX_IS_LITTLE_ENDIAN)
814  MemoryTools::XOR(msg, 0, state, 0, Rate);
815 #else
816  for (i = 0; i < (Rate >> 3); ++i)
817  {
818  state[i] ^= IntegerTools::LeBytesTo64(msg, (8 * i));
819  }
820 #endif
821 
822  blkcnt = OutLength / Rate;
823  SqueezeR24(state, Output, OutOffset, blkcnt, Rate);
824  OutOffset += blkcnt * Rate;
825  OutLength -= blkcnt * Rate;
826 
827  if (OutLength != 0)
828  {
829  PermuteR24P1600U(state);
830 
831  const size_t FNLBLK = (OutLength % sizeof(ulong) == 0) ? OutLength / sizeof(ulong) : OutLength / sizeof(ulong) + 1;
832 
833  for (i = 0; i < FNLBLK; i++)
834  {
835  IntegerTools::Le64ToBytes(state[i], msg, (8 * i));
836  }
837 
838  MemoryTools::Copy(msg, 0, Output, OutOffset, OutLength);
839  }
