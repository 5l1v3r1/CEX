201  template<typename ArrayU8>
202  static void AbsorbR24(const ArrayU8 &Input, size_t InOffset, size_t InLength, size_t Rate, byte Domain, std::array<ulong, KECCAK_STATE_SIZE> &State)
203  {
204  std::array<byte, 200> msg = { 0 };
205 
206  while (InLength >= Rate)
207  {
208  Keccak::FastAbsorb(Input, InOffset, Rate, State);
209 
210 #if defined(CEX_DIGEST_COMPACT)
211  Keccak::PermuteR24P1600C(State);
212 #else
213  Keccak::PermuteR24P1600U(State);
214 #endif
215 
216  InLength -= Rate;
217  InOffset += Rate;
218  }
219 
220  MemoryTools::Copy(Input, InOffset, msg, 0, InLength);
221  msg[InLength] = Domain;
222  msg[Rate - 1] |= 128;
223 
224 #if defined(CEX_IS_LITTLE_ENDIAN)
225  MemoryTools::XOR(msg, 0, State, 0, Rate);
226 #else
227  for (i = 0; i < (Rate >> 3); ++i)
228  {
229  State[i] ^= IntegerTools::LeBytesTo64(msg, (8 * i));
230  }
231 #endif
232  }

290  template<typename ArrayU8>
291  static void CXOFP1600(const ArrayU8 &Key, const ArrayU8 &Customization, const ArrayU8 &Name, ArrayU8 &Output, size_t OutOffset, size_t OutLength, size_t Rate)
292  {
293  static const size_t BUFFER_SIZE = KECCAK_STATE_SIZE * sizeof(ulong);
294  std::array<ulong, KECCAK_STATE_SIZE> state = { 0 };
295  std::array<byte, BUFFER_SIZE> pad = { 0 };
296  size_t i;
297  size_t offset;
298 
299  offset = Keccak::LeftEncode(pad, 0, static_cast<ulong>(Rate));
300  offset += Keccak::LeftEncode(pad, offset, static_cast<ulong>(Name.size()) * 8);
301 
302  if (Name.size() != 0)
303  {
304  for (i = 0; i < Name.size(); ++i)
305  {
306  if (offset == Rate)
307  {
308  FastAbsorb(pad, 0, Rate, state);
309  Permute(state);
310  offset = 0;
311  }
312 
313  pad[offset] = Name[i];
314  ++offset;
315  }
316  }
317 
318  offset += Keccak::LeftEncode(pad, offset, static_cast<ulong>(Customization.size()) * 8);
319 
320  if (Customization.size() != 0)
321  {
322  for (i = 0; i < Customization.size(); ++i)
323  {
324  if (offset == Rate)
325  {
326  FastAbsorb(pad, 0, Rate, state);
327  Permute(state);
328  offset = 0;
329  }
330 
331  pad[offset] = Customization[i];
332  ++offset;
333  }
334  }
335 
336  MemoryTools::Clear(pad, offset, BUFFER_SIZE - offset);
337  offset = (offset % sizeof(ulong) == 0) ? offset : offset + (sizeof(ulong) - (offset % sizeof(ulong)));
338  MemoryTools::XOR(pad, 0, state, 0, offset);
339  Permute(state);
340 
341  Keccak::Absorb(Key, 0, Key.size(), Rate, Keccak::KECCAK_CSHAKE_DOMAIN, state);
342 
343  while (OutLength != 0)
344  {
345  const size_t DIFF = IntegerTools::Min(Rate, OutLength);
346 
347  Permute(state);
348  MemoryTools::Copy(state, 0, Output, OutOffset, DIFF);
349  OutOffset += DIFF;
350  OutLength -= DIFF;
351  }
352  }

439  template<typename ArrayU8>
440  static void FastAbsorb(const ArrayU8 &Input, size_t InOffset, size_t InLength, std::array<ulong, KECCAK_STATE_SIZE> &State)
441  {
442  CEXASSERT(InLength % sizeof(ulong) == 0, "The input length is not 64-bit aligned");
443 
444 #if defined(CEX_IS_LITTLE_ENDIAN)
445  MemoryTools::XOR(Input, InOffset, State, 0, InLength);
446 #else
447  for (size_t i = 0; i < InLength / sizeof(ulong); ++i)
448  {
449  State[i] ^= IntegerTools::LeBytesTo64(Input, InOffset + (i * sizeof(ulong)));
450  }
451 #endif
452  }

463  template<typename ArrayU8>
464  static ulong LeftEncode(ArrayU8 &Output, size_t Offset, ulong Value)
465  {
466  ulong i;
467  ulong n;
468  ulong v;
469 
470  for (v = Value, n = 0; v && (n < sizeof(ulong)); ++n, v >>= 8)
471  {
472  }
473 
474  if (n == 0)
475  {
476  n = 1;
477  }
478 
479  for (i = 1; i <= n; ++i)
480  {
481  Output[Offset + i] = static_cast<uint8_t>(Value >> (8 * (n - i)));
482  }
483 
484  Output[Offset] = static_cast<uint8_t>(n);
485 
486  return (n + 1);
487  }

591  template<typename ArrayU8A, typename ArrayU8B>
592  static void XOFP1600(const ArrayU8A &Input, size_t InOffset, size_t InLength, ArrayU8B &Output, size_t OutOffset, size_t OutLength, size_t Rate)
593  {
594  std::array<byte, KECCAK_STATE_SIZE * sizeof(ulong)> msg = { 0 };
595  std::array<ulong, KECCAK_STATE_SIZE> state = { 0 };
596  size_t blkcnt;
597  size_t i;
598 
599  while (InLength >= Rate)
600  {
601  FastAbsorb(Input, InOffset, Rate, state);
602  Permute(state);
603  InLength -= Rate;
604  InOffset += Rate;
605  }
606 
607  MemoryTools::Copy(Input, InOffset, msg, 0, InLength);
608  msg[InLength] = KECCAK_SHAKE_DOMAIN;
609  MemoryTools::Clear(msg, InLength + 1, Rate - InLength + 1);
610  msg[Rate - 1] |= 128;
611 
612 #if defined(CEX_IS_LITTLE_ENDIAN)
613  MemoryTools::XOR(msg, 0, state, 0, Rate);
614 #else
615  for (i = 0; i < (Rate >> 3); ++i)
616  {
617  state[i] ^= IntegerTools::LeBytesTo64(msg, (8 * i));
618  }
619 #endif
620 
621  blkcnt = OutLength / Rate;
622  Squeeze(state, Output, OutOffset, blkcnt, Rate);
623  OutOffset += blkcnt * Rate;
624  OutLength -= blkcnt * Rate;
625 
626  if (OutLength != 0)
627  {
628  Permute(state);
629 
630  const size_t FNLBLK = (OutLength % sizeof(ulong) == 0) ? OutLength / sizeof(ulong) : OutLength / sizeof(ulong) + 1;
631 
632  for (i = 0; i < FNLBLK; i++)
633  {
634  IntegerTools::Le64ToBytes(state[i], msg, (8 * i));
635  }
636 
637  MemoryTools::Copy(msg, 0, Output, OutOffset, OutLength);
638  }
639  }
