337  inline static void Mul64x64To128(ulong X, ulong Y, ulong* Low, ulong* High)
338  {
339 #if defined(CEX_FAST_64X64_MUL)
340  CEX_FAST_64X64_MUL(X, Y, Low, High);
341 #else
342 
343  // Do a 64x64->128 multiply using four 32x32->64 multiplies plus
344  // some adds and shifts. Last resort for CPUs like UltraSPARC (with
345  // 64-bit registers/ALU, but no 64x64->128 multiply) or 32-bit CPUs.
346  const size_t HWORD_BITS = 32;
347  const uint HWORD_MASK = 0xFFFFFFFFUL;
348 
349  const uint ah = (X >> HWORD_BITS);
350  const uint al = (X & HWORD_MASK);
351  const uint bh = (Y >> HWORD_BITS);
352  const uint bl = (Y & HWORD_MASK);
353 
354  ulong x0 = static_cast<ulong>(ah) * bh;
355  ulong x1 = static_cast<ulong>(al) * bh;
356  ulong x2 = static_cast<ulong>(ah) * bl;
357  ulong x3 = static_cast<ulong>(al) * bl;
358 
359  // this cannot overflow as (2^32-1)^2 + 2^32-1 < 2^64-1
360  x2 += x3 >> HWORD_BITS;
361  // this one can overflow
362  x2 += x1;
363  // propagate the carry if any
364  x0 += static_cast<ulong>(static_cast<bool>(x2 < x1)) << HWORD_BITS;
365 
366  *High = x0 + (x2 >> HWORD_BITS);
367  *Low = ((x2 & HWORD_MASK) << HWORD_BITS) + (x3 & HWORD_MASK);
368 #endif
369  }
