327  inline static void Mul64x64To128(ulong X, ulong Y, ulong* Low, ulong* High)
328  {
329 #if defined(CEX_FAST_64X64_MUL)
330  CEX_FAST_64X64_MUL(X, Y, Low, High);
331 #else
332 
333  // Do a 64x64->128 multiply using four 32x32->64 multiplies plus
334  // some adds and shifts. Last resort for CPUs like UltraSPARC (with
335  // 64-bit registers/ALU, but no 64x64->128 multiply) or 32-bit CPUs.
336  const size_t HWORD_BITS = 32;
337  const uint HWORD_MASK = 0xFFFFFFFFUL;
338 
339  const uint ah = (X >> HWORD_BITS);
340  const uint al = (X & HWORD_MASK);
341  const uint bh = (Y >> HWORD_BITS);
342  const uint bl = (Y & HWORD_MASK);
343 
344  ulong x0 = static_cast<ulong>(ah) * bh;
345  ulong x1 = static_cast<ulong>(al) * bh;
346  ulong x2 = static_cast<ulong>(ah) * bl;
347  ulong x3 = static_cast<ulong>(al) * bl;
348 
349  // this cannot overflow as (2^32-1)^2 + 2^32-1 < 2^64-1
350  x2 += x3 >> HWORD_BITS;
351  // this one can overflow
352  x2 += x1;
353  // propagate the carry if any
354  x0 += static_cast<ulong>(static_cast<bool>(x2 < x1)) << HWORD_BITS;
355 
356  *High = x0 + (x2 >> HWORD_BITS);
357  *Low = ((x2 & HWORD_MASK) << HWORD_BITS) + (x3 & HWORD_MASK);
358 #endif
359  }
