68  static void Round256W(T &A, T &B, T &C, T &D, T &E, T &F, T &G, T &H, T &M, T &P)
69  {
70  T R(H + (((E >> 6) | (E << 26)) ^ ((E >> 11) | (E << 21)) ^ ((E >> 25) | (E << 7))) + ((E & F) ^ (~E & G)) + M + P);
71  D += R;
72  H = R + ((((A >> 2) | (A << 30)) ^ ((A >> 13) | (A << 19)) ^ ((A >> 22) | (A << 10))) + ((A & B) ^ (A & C) ^ (B & C)));
73  }

132  template<typename ArrayU8>
133  static void Compute256(const ArrayU8 &Input, size_t InOffset, size_t InLength, ArrayU8 &Output, size_t OutOffset)
134  {
135  std::array<uint, 8> state = { 0 };
136  std::vector<byte> buf(SHA256_RATE_SIZE);
137  ulong bitlen;
138  ulong t;
139 
140  t = 0;
141  MemoryTools::Copy(SHA2::SHA256State, 0, state, 0, state.size() * sizeof(uint));
142 
143  while (InLength >= SHA256_RATE_SIZE)
144  {
145  PermuteR64P512U(Input, InOffset, state);
146  InLength -= SHA256_RATE_SIZE;
147  InOffset += SHA256_RATE_SIZE;
148  t += SHA256_RATE_SIZE;
149  }
150 
151  t += InLength;
152  bitlen = (t << 3);
153  MemoryTools::Copy(Input, InOffset, buf, 0, InLength);
154 
155  buf[InLength] = 128;
156  ++InLength;
157 
158  if (InLength > 56)
159  {
160  PermuteR64P512U(buf, 0, state);
161  MemoryTools::Clear(buf, 0, SHA2::SHA256_RATE_SIZE);
162  }
163 
164  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen) >> 32), buf, 56);
165  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen)), buf, 60);
166 
167  PermuteR64P512U(buf, 0, state);
168 
169  // copy as big endian aligned to output code
170  IntegerTools::BeUL256ToBlock(state, 0, Output, OutOffset);
171  }

184  template<typename ArrayU8>
185  static void MACR64P512(const ArrayU8 &Key, const ArrayU8 &Input, size_t InOffset, size_t InLength, ArrayU8 &Output, size_t OutOffset)
186  {
187  CEXASSERT(Key.size() <= SHA256_RATE_SIZE, "The Mac key array must be a maximum of 64 bytes in length");
188 
189  const byte IPAD = 0x36;
190  const byte OPAD = 0x5C;
191  std::vector<byte> buf(SHA256_RATE_SIZE);
192  std::vector<byte> ipad(SHA256_RATE_SIZE);
193  std::vector<byte> opad(SHA256_RATE_SIZE);
194  std::array<uint, 8> state = { 0 };
195  ulong bitlen;
196  ulong t;
197 
198  // copy in the key and xor the hamming weights into input and output pads
199  MemoryTools::Copy(Key, 0, ipad, 0, Key.size());
200  MemoryTools::Copy(ipad, 0, opad, 0, ipad.size());
201  MemoryTools::XorPad(ipad, IPAD);
202  MemoryTools::XorPad(opad, OPAD);
203 
204  // initialize the sha256 state
205  MemoryTools::Copy(SHA2::SHA256State, 0, state, 0, state.size() * sizeof(uint));
206 
207  // permute the input pad
208  PermuteR64P512U(ipad, 0, state);
209  t = SHA256_RATE_SIZE;
210 
211  // process the message
212  while (InLength >= SHA256_RATE_SIZE)
213  {
214  PermuteR64P512U(Input, InOffset, state);
215  InLength -= SHA256_RATE_SIZE;
216  InOffset += SHA256_RATE_SIZE;
217  t += SHA256_RATE_SIZE;
218  }
219 
220  // finalize the message data
221  t += InLength;
222  bitlen = (t << 3);
223  MemoryTools::Copy(Input, InOffset, buf, 0, InLength);
224  buf[InLength] = 128;
225  ++InLength;
226 
227  if (InLength > 56)
228  {
229  PermuteR64P512U(buf, 0, state);
230  MemoryTools::Clear(buf, 0, SHA2::SHA256_RATE_SIZE);
231  }
232 
233  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen) >> 32), buf, 56);
234  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen)), buf, 60);
235  PermuteR64P512U(buf, 0, state);
236 
237  // store the code in the buffer
238  IntegerTools::BeUL256ToBlock(state, 0, buf, 0);
239  MemoryTools::Clear(buf, SHA256_DIGEST_SIZE, SHA256_DIGEST_SIZE);
240  // reset the sha2 state
241  MemoryTools::Copy(SHA2::SHA256State, 0, state, 0, state.size() * sizeof(uint));
242 
243  // permute the output pad
244  PermuteR64P512U(opad, 0, state);
245  // finalize the buffer
246  t = SHA256_RATE_SIZE + SHA256_DIGEST_SIZE;
247  bitlen = (t << 3);
248  buf[SHA256_DIGEST_SIZE] = 128;
249  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen) >> 32), buf, 56);
250  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen)), buf, 60);
251  PermuteR64P512U(buf, 0, state);
252 
253  // copy as big endian aligned to output code
254  IntegerTools::BeUL256ToBlock(state, 0, Output, OutOffset);
255  }
