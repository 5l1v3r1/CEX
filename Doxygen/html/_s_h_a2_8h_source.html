134  template<typename ArrayU8>
135  static void Compute256(const ArrayU8 &Input, size_t InOffset, size_t InLength, ArrayU8 &Output, size_t OutOffset)
136  {
137  std::array<uint, 8> state = { 0 };
138  std::vector<byte> buf(SHA256_RATE_SIZE);
139  ulong bitlen;
140  ulong t;
141 
142  t = 0;
143  MemoryTools::Copy(SHA256State, 0, state, 0, state.size() * sizeof(uint));
144 
145  while (InLength >= SHA256_RATE_SIZE)
146  {
147  PermuteR64P512U(Input, InOffset, state);
148  InLength -= SHA256_RATE_SIZE;
149  InOffset += SHA256_RATE_SIZE;
150  t += SHA256_RATE_SIZE;
151  }
152 
153  t += InLength;
154  bitlen = (t << 3);
155  MemoryTools::Copy(Input, InOffset, buf, 0, InLength);
156 
157  buf[InLength] = 128;
158  ++InLength;
159 
160  if (InLength > 56)
161  {
162  PermuteR64P512U(buf, 0, state);
163  MemoryTools::Clear(buf, 0, SHA256_RATE_SIZE);
164  }
165 
166  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen) >> 32), buf, 56);
167  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen)), buf, 60);
168 
169  PermuteR64P512U(buf, 0, state);
170 
171  // copy as big endian aligned to output code
172  IntegerTools::BeUL256ToBlock(state, 0, Output, OutOffset);
173  }

186  template<typename ArrayU8>
187  static void MACR64P512(const ArrayU8 &Key, const ArrayU8 &Input, size_t InOffset, size_t InLength, ArrayU8 &Output, size_t OutOffset)
188  {
189  CEXASSERT(Key.size() <= SHA256_RATE_SIZE, "The Mac key array must be a maximum of 64 bytes in length");
190 
191  const byte IPAD = 0x36;
192  const byte OPAD = 0x5C;
193  std::vector<byte> buf(SHA256_RATE_SIZE);
194  std::vector<byte> ipad(SHA256_RATE_SIZE);
195  std::vector<byte> opad(SHA256_RATE_SIZE);
196  std::array<uint, 8> state = { 0 };
197  ulong bitlen;
198  ulong t;
199 
200  // copy in the key and xor the hamming weights into input and output pads
201  MemoryTools::Copy(Key, 0, ipad, 0, Key.size());
202  MemoryTools::Copy(ipad, 0, opad, 0, ipad.size());
203  MemoryTools::XorPad(ipad, IPAD);
204  MemoryTools::XorPad(opad, OPAD);
205 
206  // initialize the sha256 state
207  MemoryTools::Copy(SHA256State, 0, state, 0, state.size() * sizeof(uint));
208 
209  // permute the input pad
210  PermuteR64P512U(ipad, 0, state);
211  t = SHA256_RATE_SIZE;
212 
213  // process the message
214  while (InLength >= SHA256_RATE_SIZE)
215  {
216  PermuteR64P512U(Input, InOffset, state);
217  InLength -= SHA256_RATE_SIZE;
218  InOffset += SHA256_RATE_SIZE;
219  t += SHA256_RATE_SIZE;
220  }
221 
222  // finalize the message data
223  t += InLength;
224  bitlen = (t << 3);
225  MemoryTools::Copy(Input, InOffset, buf, 0, InLength);
226  buf[InLength] = 128;
227  ++InLength;
228 
229  if (InLength > 56)
230  {
231  PermuteR64P512U(buf, 0, state);
232  MemoryTools::Clear(buf, 0, SHA256_RATE_SIZE);
233  }
234 
235  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen) >> 32), buf, 56);
236  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen)), buf, 60);
237  PermuteR64P512U(buf, 0, state);
238 
239  // store the code in the buffer
240  IntegerTools::BeUL256ToBlock(state, 0, buf, 0);
241  MemoryTools::Clear(buf, SHA256_DIGEST_SIZE, SHA256_DIGEST_SIZE);
242  // reset the sha2 state
243  MemoryTools::Copy(SHA256State, 0, state, 0, state.size() * sizeof(uint));
244 
245  // permute the output pad
246  PermuteR64P512U(opad, 0, state);
247  // finalize the buffer
248  t = SHA256_RATE_SIZE + SHA256_DIGEST_SIZE;
249  bitlen = (t << 3);
250  buf[SHA256_DIGEST_SIZE] = 128;
251  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen) >> 32), buf, 56);
252  IntegerTools::Be32ToBytes(static_cast<uint>(static_cast<ulong>(bitlen)), buf, 60);
253  PermuteR64P512U(buf, 0, state);
254 
255  // copy as big endian aligned to output code
256  IntegerTools::BeUL256ToBlock(state, 0, Output, OutOffset);
257  }
