190  {
191  uint8_t buf1[2 * SPX_N];
192  uint8_t buf2[SPX_N + SPX_ADDR_BYTES + 2 * SPX_N];
193  uint8_t mask[2 * SPX_N];
194  size_t i;

196  // If leafidx is odd (last bit = 1), current path element is a right child
197  // and authpath has to go left. Otherwise it is the other way around.
198  if (leafidx & 1)
199  {
200  memcpy(buf1 + SPX_N, leaf, SPX_N);
201  memcpy(buf1, authpath, SPX_N);
202  }

206  memcpy(buf1 + SPX_N, authpath, SPX_N);

210

212  {

214  idxoffset >>= 1;
215  // Set the address of the node we're creating.
216  SphincsUtils::set_tree_height(addr, i + 1);
217  SphincsUtils::set_tree_index(addr, leafidx + idxoffset);
218

220

233  }

236  leafidx >>= 1;

384  // Computes the message hash using R, the public key, and the message.
385  // Notably, it requires m to have SPX_N + SPX_PK_BYTES bytes of space available
386  // in front of the pointer, i.e. before the message, to use for the prefix.
387  // This is necessary to prevent having to move the * message around (and
388  // allocate memory for it).
389  // Outputs the message digest and the index of the leaf. The index is split in
390  // the tree index and the leaf index, for convenient copying to an address.
391  static void hash_message(uint8_t* digest, ulong* tree, uint* leaf_idx, const uint8_t* R, const uint8_t* pk, uint8_t* m, ulong mlen)
392  {
393  static const int32_t SPX_TREE_BITS = (SPX_TREE_HEIGHT * (SPX_D - 1));
394  static const int32_t SPX_TREE_BYTES = ((SPX_TREE_BITS + 7) / 8);
395  static const int32_t SPX_LEAF_BITS = SPX_TREE_HEIGHT;
396  static const int32_t SPX_LEAF_BYTES = ((SPX_LEAF_BITS + 7) / 8);
397  static const int32_t SPX_DGST_BYTES = (SPX_FORS_MSG_BYTES + SPX_TREE_BYTES + SPX_LEAF_BYTES);
398 
399  uint8_t buf[SPX_DGST_BYTES];
400  uint8_t *bufp = buf;
401 
402  memcpy(m - SPX_N - SPX_PK_BYTES, R, SPX_N);
403  memcpy(m - SPX_PK_BYTES, pk, SPX_PK_BYTES);
404  shake256(buf, SPX_DGST_BYTES, m - SPX_N - SPX_PK_BYTES, mlen + SPX_N + SPX_PK_BYTES);
405  memcpy(digest, bufp, SPX_FORS_MSG_BYTES);
406  bufp += SPX_FORS_MSG_BYTES;
407 
408 #if SPX_TREE_BITS > 64
409 #error For given height and depth, 64 bits cannot represent all subtrees
410 #endif
411 
412  *tree = SphincsUtils::bytes_to_ull(bufp, SPX_TREE_BYTES);
413  *tree &= (~(ulong)0) >> (64 - SPX_TREE_BITS);
414  bufp += SPX_TREE_BYTES;
415  *leaf_idx = SphincsUtils::bytes_to_ull(bufp, SPX_LEAF_BYTES);
416  *leaf_idx &= (~(uint)0) >> (32 - SPX_LEAF_BITS);
417  }

481  // For a given leaf index, computes the authentication path and the resulting
482  // root node using Merkle's TreeHash algorithm.
483  // Expects the layer and tree parts of the tree_addr to be set, as well as the
484  // tree type (i.e. SPX_ADDR_TYPE_HASHTREE or SPX_ADDR_TYPE_FORSTREE).
485  // Applies the offset idx_offset to indices before building addresses, so that
486  // it is possible to continue counting indices across trees.
487  static void treehash(uint8_t* root, uint8_t* authpath, const uint8_t* skseed, const uint8_t* pkseed, uint leafidx,
488  uint idxoffset, uint treeheight, uint treeaddr[8], uint8_t* stack, uint* heights,
489  void(*gen_leaf)(uint8_t*, const uint8_t*, const uint8_t*, uint, const uint[8]))
490  {
491  uint8_t buf[SPX_N + SPX_ADDR_BYTES + 2 * SPX_N];
492  uint8_t mask[2 * SPX_N];
493  uint offset;
494  uint idx;
495  uint treeidx;
496 
497  offset = 0;
498 
499  for (idx = 0; idx < (uint)(1 << treeheight); idx++)
500  {
501  // Add the next leaf node to the stack.
502  gen_leaf(stack + (offset * SPX_N), skseed, pkseed, idx + idxoffset, treeaddr);
503  offset++;
504  heights[offset - 1] = 0;
505 
506  // If this is a node we need for the auth path..
507  if ((leafidx ^ 0x1) == idx)
508  {
509  memcpy(authpath, stack + ((offset - 1) * SPX_N), SPX_N);
510  }
511 
512  // While the top-most nodes are of equal height..
513  while (offset >= 2 && heights[offset - 1] == heights[offset - 2])
514  {
515  // Compute index of the new node, in the next layer.
516  treeidx = (idx >> (heights[offset - 1] + 1));
517 
518  // Set the address of the node we're creating.
519  SphincsUtils::set_tree_height(treeaddr, heights[offset - 1] + 1);
520  SphincsUtils::set_tree_index(treeaddr, treeidx + (idxoffset >> (heights[offset - 1] + 1)));
521  // Hash the top-most nodes from the stack together.
522  thash(stack + ((offset - 2) * SPX_N), stack + ((offset - 2) * SPX_N), 2, pkseed, treeaddr, buf, mask);
523  offset--;
524  // Note that the top-most node is now one layer higher.
525  heights[offset - 1]++;
526 
527  // If this is a node we need for the auth path..
528  if (((leafidx >> heights[offset - 1]) ^ 0x1) == treeidx)
529  {
530  memcpy(authpath + (heights[offset - 1] * SPX_N), stack + ((offset - 1) * SPX_N), SPX_N);
531  }
532  }
533  }
534 
535  memcpy(root, stack, SPX_N);
536  }

699  // preparation or computation it needs, based on the public seed.
700  initialize_hash_function(pkseed, skseed);

703  SphincsUtils::set_type(treeaddr, SPX_ADDR_TYPE_HASHTREE);

706  // things when computing the hash over the message.
707  // We need to do this from back to front, so that it works when sm = m
708  for (i = mlen; i > 0; i--)

711  }

715  // This can help counter side-channel attacks that would benefit from
716  // getting a large number of traces when the signer uses the same nodes.
717  if (gen_random(optrand, SPX_N) != 1)
718  {
719  return 0;
720  }
721  memset(optrand, 0, sizeof(optrand));

735

738  SphincsUtils::set_layer_addr(treeaddr, i);
739  SphincsUtils::set_tree_addr(treeaddr, tree);
740

753  idxleaf = (tree & ((1 << SPX_TREE_HEIGHT) - 1));

755  }

782  // preparation or computation it needs, based on the public seed.

785  SphincsUtils::set_type(treeaddr, SPX_ADDR_TYPE_HASHTREE);

789  // but SPHINCS+ signatures are always exactly SPX_BYTES.
