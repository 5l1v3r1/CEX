123  static void Swap(Array &X, size_t XOffset, Array &Y, size_t YOffset, size_t Length, int32_t Mask)
124  {
125  Array::value_type c;
126  Array::value_type t;
127  Array::value_type xi;
128  Array::value_type yi;
129  size_t i;
130 
131  c = Mask;
132 
133  for (i = 0; i < Length; ++i)
134  {
135  xi = X[XOffset + i];
136  yi = Y[YOffset + i];
137  t = c & (xi ^ yi);
138  xi ^= t;
139  yi ^= t;
140  X[XOffset + i] = xi;
141  Y[YOffset + i] = yi;
142  }
143  }
