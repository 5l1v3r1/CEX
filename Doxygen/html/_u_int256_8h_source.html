401  inline static void Transpose(UInt256 &X0, UInt256 &X1, UInt256 &X2, UInt256 &X3)
402  {
403  __m256i T0 = _mm256_unpacklo_epi32(X0.ymm, X1.ymm);
404  __m256i T1 = _mm256_unpacklo_epi32(X2.ymm, X3.ymm);
405  __m256i T2 = _mm256_unpackhi_epi32(X0.ymm, X1.ymm);
406  __m256i T3 = _mm256_unpackhi_epi32(X2.ymm, X3.ymm);
407  X0.ymm = _mm256_unpacklo_epi64(T0, T1);
408  X1.ymm = _mm256_unpackhi_epi64(T0, T1);
409  X2.ymm = _mm256_unpacklo_epi64(T2, T3);
410  X3.ymm = _mm256_unpackhi_epi64(T2, T3);
411  }
