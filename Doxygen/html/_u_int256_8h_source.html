406  inline static void Transpose(UInt256 &X0, UInt256 &X1, UInt256 &X2, UInt256 &X3)
407  {
408  __m256i T0 = _mm256_unpacklo_epi32(X0.ymm, X1.ymm);
409  __m256i T1 = _mm256_unpacklo_epi32(X2.ymm, X3.ymm);
410  __m256i T2 = _mm256_unpackhi_epi32(X0.ymm, X1.ymm);
411  __m256i T3 = _mm256_unpackhi_epi32(X2.ymm, X3.ymm);
412  X0.ymm = _mm256_unpacklo_epi64(T0, T1);
413  X1.ymm = _mm256_unpackhi_epi64(T0, T1);
414  X2.ymm = _mm256_unpacklo_epi64(T2, T3);
415  X3.ymm = _mm256_unpackhi_epi64(T2, T3);
416  }
