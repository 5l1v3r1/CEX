96  CexAssert(ELMLEN <= Length, "Integer type is larger than length");

98  size_t prcCtr = 0;
99 
100 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
101 # if defined(__AVX512__)
102  const size_t SMDBLK = 64 / ELMLEN;
103 # elif defined(__AVX2__)
104  const size_t SMDBLK = 32 / ELMLEN;
105 # else
106  const size_t SMDBLK = 16 / ELMLEN;
107 # endif
108 
109  if (Length / ELMLEN >= SMDBLK)
110  {
111  const size_t ALNLEN = (Length / (SMDBLK * ELMLEN)) * SMDBLK;
112 
113  while (prcCtr != ALNLEN)
114  {
115 #if defined(__AVX512__)
116  CLEAR512(Output, Offset + prcCtr);
117 #elif defined(__AVX2__)
118  CLEAR256(Output, Offset + prcCtr);
119 #elif defined(__AVX__)
120  CLEAR128(Output, Offset + prcCtr);
121 #endif
122  prcCtr += SMDBLK;
123  }
124  }
125 #endif
126 
127  if (prcCtr * ELMLEN != Length)
128  {
129  std::memset(&Output[Offset + prcCtr], 0, Length - (prcCtr * ELMLEN));

198  inline static bool Compare(const Array &A, size_t AOffset, const Array &B, size_t BOffset, size_t Elements)
199  {
200  CexAssert((A.size() - AOffset) >= Length, "Length is larger than A size");
201  CexAssert((B.size() - BOffset) >= Length, "Length is larger than B size");
202 
203  size_t diff = 0;
204 
205  for (size_t i = 0; i != Elements; ++i)
206  {
207  diff |= (A[AOffset + i] ^ B[BOffset + i]);
208  }
209 
210  return diff == 0;

321  template <typename ArrayA, typename ArrayB>
322  inline static void Copy(const ArrayA &Input, size_t InOffset, ArrayB &Output, size_t OutOffset, size_t Length)
323  {
324  if (Length != 0)
325  {
326  const size_t INPLEN = sizeof(ArrayA::value_type);
327  const size_t OUTLEN = sizeof(ArrayB::value_type);
328 
329  CexAssert((Input.size() - InOffset) * INPLEN >= Length, "Length is larger than input capacity");
330  CexAssert((Output.size() - OutOffset) * OUTLEN >= Length, "Length is larger than output capacity");
331  CexAssert(INPLEN <= 64 && OUTLEN <= 64, "Integer type is larger than 128 bits");
332 
333 #if defined(__AVX512__)
334  const size_t SMDBLK = 64;
335 #elif defined(__AVX2__)
336  const size_t SMDBLK = 32;
337 #elif defined(__AVX__)
338  const size_t SMDBLK = 16;
339 #endif
340 
341  size_t prcCtr = 0;
342 
343 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
344  if (Length >= SMDBLK)
345  {
346  const size_t ALNLEN = Length - (Length % SMDBLK);
347 
348  while (prcCtr != ALNLEN)
349  {
350 #if defined(__AVX512__)
351  COPY512(Input, InOffset + (prcCtr / INPLEN), Output, OutOffset + (prcCtr / OUTLEN));
352 #elif defined(__AVX2__)
353  COPY256(Input, InOffset + (prcCtr / INPLEN), Output, OutOffset + (prcCtr / OUTLEN));
354 #elif defined(__AVX__)
355  COPY128(Input, InOffset + (prcCtr / INPLEN), Output, OutOffset + (prcCtr / OUTLEN));
356 #endif
357  prcCtr += SMDBLK;
358  }
359  }
360 #endif
361 
362  if (prcCtr != Length)
363  {
364  std::memcpy(&Output[OutOffset + (prcCtr / OUTLEN)], &Input[InOffset + (prcCtr / INPLEN)], Length - prcCtr);
365  }
366  }
367  }
