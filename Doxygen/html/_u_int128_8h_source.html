388  inline static void Transpose(UInt128 &X0, UInt128 &X1, UInt128 &X2, UInt128 &X3)
389  {
390  __m128i T0 = _mm_unpacklo_epi32(X0.xmm, X1.xmm);
391  __m128i T1 = _mm_unpacklo_epi32(X2.xmm, X3.xmm);
392  __m128i T2 = _mm_unpackhi_epi32(X0.xmm, X1.xmm);
393  __m128i T3 = _mm_unpackhi_epi32(X2.xmm, X3.xmm);
394  X0.xmm = _mm_unpacklo_epi64(T0, T1);
395  X1.xmm = _mm_unpackhi_epi64(T0, T1);
396  X2.xmm = _mm_unpacklo_epi64(T2, T3);
397  X3.xmm = _mm_unpackhi_epi64(T2, T3);
398  }
