390  inline static void Transpose(UInt128 &X0, UInt128 &X1, UInt128 &X2, UInt128 &X3)
391  {
392  __m128i T0 = _mm_unpacklo_epi32(X0.xmm, X1.xmm);
393  __m128i T1 = _mm_unpacklo_epi32(X2.xmm, X3.xmm);
394  __m128i T2 = _mm_unpackhi_epi32(X0.xmm, X1.xmm);
395  __m128i T3 = _mm_unpackhi_epi32(X2.xmm, X3.xmm);
396  X0.xmm = _mm_unpacklo_epi64(T0, T1);
397  X1.xmm = _mm_unpackhi_epi64(T0, T1);
398  X2.xmm = _mm_unpacklo_epi64(T2, T3);
399  X3.xmm = _mm_unpackhi_epi64(T2, T3);
400  }
