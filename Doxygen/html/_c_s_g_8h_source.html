327  template<typename ArrayA, typename ArrayB>
328  static void AbsorbBlock(const ArrayA &Input, size_t InOffset, size_t Length, ArrayB &State)
329  {
330  for (size_t i = 0; i < Length / sizeof(ulong); ++i)
331  {
332  State[i] ^= static_cast<ulong>(IntUtils::LeBytesTo64(Input, InOffset + (i * sizeof(ulong))));
333  }
334  }

336  template<typename Array>
337  static size_t LeftEncode(Array &Buffer, size_t Offset, size_t Value)
338  {
339  size_t i;
340  size_t n;
341  size_t v;
342 
343  for (v = Value, n = 0; v && (n < sizeof(size_t)); ++n, v >>= 8);
344 
345  if (n == 0)
346  {
347  n = 1;
348  }
349 
350  for (i = 1; i <= n; ++i)
351  {
352  Buffer[Offset + i] = (uint8_t)(Value >> (8 * (n - i)));
353  }
354 
355  Buffer[Offset] = (uint8_t)n;
356 
357  return (n + 1);
358  }
