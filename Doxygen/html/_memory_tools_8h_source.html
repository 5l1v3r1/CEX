90  template <typename Array>
91  inline static void PrefetchL1(Array &Input, size_t Offset, size_t Length)
92  {
93  const size_t ELMLEN = sizeof(Array::value_type);
94  CEXASSERT((Input.size() - Offset) * ELMLEN >= Length, "Length is larger than output capacity");
95  CEXASSERT(ELMLEN <= Length, "Integer type is larger than length");
96 
97 #if defined(__AVX__)
98  PREFETCHT1(Input.data() + (Offset * ELMLEN), Length);
99 #else
100  volatile Array::value_type tmp = 0;
101 
102  for (size_t i = 0; i < Length / ELMLEN; ++i)
103  {
104  tmp ^= Input[i + Offset];
105  }
106 #endif
107  }

158  template <typename Array>
159  inline static void Clear(Array &Output, size_t Offset, size_t Length)
160  {
161  if (Length != 0)
162  {
163  const size_t ELMLEN = sizeof(Array::value_type);
164  CEXASSERT((Output.size() - Offset) * ELMLEN >= Length, "Length is larger than output capacity");
165  CEXASSERT(ELMLEN <= Length, "Integer type is larger than length");
166 
167  size_t prcCtr = 0;
168 
169 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
170 # if defined(__AVX512__)
171  const size_t SMDBLK = 64 / ELMLEN;
172 # elif defined(__AVX2__)
173  const size_t SMDBLK = 32 / ELMLEN;
174 # else
175  const size_t SMDBLK = 16 / ELMLEN;
176 # endif
177 
178  if (Length / ELMLEN >= SMDBLK)
179  {
180  const size_t ALNLEN = (Length / (SMDBLK * ELMLEN)) * SMDBLK;
181 
182  while (prcCtr != ALNLEN)
183  {
184 #if defined(__AVX512__)
185  CLEAR512(Output, Offset + prcCtr);
186 #elif defined(__AVX2__)
187  CLEAR256(Output, Offset + prcCtr);
188 #elif defined(__AVX__)
189  CLEAR128(Output, Offset + prcCtr);
190 #endif
191  prcCtr += SMDBLK;
192  }
193  }
194 #endif
195 
196  if (prcCtr * ELMLEN != Length)
197  {
198  std::memset(&Output[Offset + prcCtr], 0, Length - (prcCtr * ELMLEN));
199  }
200  }
201  }

270  inline static bool Compare(const Array &A, size_t AOffset, const Array &B, size_t BOffset, size_t Elements)
271  {
272  CEXASSERT((A.size() - AOffset) >= Length, "Length is larger than A size");
273  CEXASSERT((B.size() - BOffset) >= Length, "Length is larger than B size");
274 
275  size_t diff = 0;
276 
277  for (size_t i = 0; i != Elements; ++i)

505  template <typename ArrayA, typename ArrayB>
506  inline static void Copy(const ArrayA &Input, size_t InOffset, ArrayB &Output, size_t OutOffset, size_t Length)
507  {
508  if (Length != 0)
509  {
510  const size_t INPLEN = sizeof(ArrayA::value_type);
511  const size_t OUTLEN = sizeof(ArrayB::value_type);
512 
513  CEXASSERT((Input.size() - InOffset) * INPLEN >= Length, "Length is larger than input capacity");
514  CEXASSERT((Output.size() - OutOffset) * OUTLEN >= Length, "Length is larger than output capacity");
515 
516 #if defined(__AVX512__)
517  const size_t SMDBLK = 64;
518 #elif defined(__AVX2__)
519  const size_t SMDBLK = 32;
520 #elif defined(__AVX__)
521  const size_t SMDBLK = 16;
522 #endif
523 
524  size_t prcCtr = 0;
525 
526 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
527  if (Length >= SMDBLK)
528  {
529  const size_t ALNLEN = Length - (Length % SMDBLK);
530 
531  while (prcCtr != ALNLEN)
532  {
533 #if defined(__AVX512__)
534  COPY512(Input, InOffset + (prcCtr / INPLEN), Output, OutOffset + (prcCtr / OUTLEN));
535 #elif defined(__AVX2__)
536  COPY256(Input, InOffset + (prcCtr / INPLEN), Output, OutOffset + (prcCtr / OUTLEN));
537 #elif defined(__AVX__)
538  COPY128(Input, InOffset + (prcCtr / INPLEN), Output, OutOffset + (prcCtr / OUTLEN));
539 #endif
540  prcCtr += SMDBLK;
541  }
542  }
543 #endif
544 
545  if (prcCtr != Length)
546  {
547  std::memcpy(&Output[OutOffset + (prcCtr / OUTLEN)], &Input[InOffset + (prcCtr / INPLEN)], Length - prcCtr);
548  }
549  }
550  }
