91  inline static void PrefetchL1(Array &Input, size_t Offset, size_t Length)
92  {
93  const size_t ELMLEN = sizeof(Array::value_type);
94 
95 #if defined(__AVX__)
96  PREFETCHT1(Input.data() + (Offset * ELMLEN), Length);

99 
100  tmp = 0;
101 
102  for (size_t i = 0; i < Length / ELMLEN; ++i)
103  {

105  }

158  template <typename Array>
159  inline static void Clear(Array &Output, size_t Offset, size_t Length)
160  {
161  size_t pctr;
162 
163  if (Length != 0)
164  {
165  const size_t ELMLEN = sizeof(Array::value_type);
166 
167  pctr = 0;
168 
169 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
170 # if defined(__AVX512__)
171  const size_t SMDBLK = 64 / ELMLEN;
172 # elif defined(__AVX2__)
173  const size_t SMDBLK = 32 / ELMLEN;
174 # else
175  const size_t SMDBLK = 16 / ELMLEN;
176 # endif
177 
178  if (Length / ELMLEN >= SMDBLK)
179  {
180  const size_t ALNLEN = (Length / (SMDBLK * ELMLEN)) * SMDBLK;
181 
182  while (pctr != ALNLEN)
183  {
184 #if defined(__AVX512__)
185  CLEAR512(Output, Offset + pctr);
186 #elif defined(__AVX2__)
187  CLEAR256(Output, Offset + pctr);
188 #elif defined(__AVX__)
189  CLEAR128(Output, Offset + pctr);
190 #endif
191  pctr += SMDBLK;
192  }
193  }
194 #endif
195 
196  if (pctr * ELMLEN != Length)
197  {
198  std::memset(&Output[Offset + pctr], 0, Length - (pctr * ELMLEN));
199  }
200  }
201  }

218  template <typename ArrayA, typename ArrayB>
219  inline static void AND(const ArrayA &Input, size_t InOffset, ArrayB &Output, size_t OutOffset, size_t Length)
220  {
221  const size_t INPLEN = sizeof(ArrayA::value_type);
222  const size_t OTPLEN = sizeof(ArrayB::value_type);
223  size_t i;
224  size_t pctr;
225 
226  CEXASSERT((Input.size() - InOffset) * INPLEN >= Length, "Length is larger than input size");
227  CEXASSERT((Output.size() - OutOffset) * OTPLEN >= Length, "Length is larger than output size");
228  CEXASSERT(Length > 0, "Length can not be zero");
229  CEXASSERT(INPLEN <= Length, "Integer type is larger than length");
230  CEXASSERT(OTPLEN <= Length, "Integer type is larger than length");
231 
232 
233  if (Length != 0)
234  {
235  const size_t ELMLEN = sizeof(ArrayA::value_type);
236  pctr = 0;
237 
238 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
239 # if defined(__AVX512__)
240  const size_t SMDBLK = 64;
241 # elif defined(__AVX2__)
242  const size_t SMDBLK = 32;
243 # else
244  const size_t SMDBLK = 16;
245 # endif
246 
247  if (Length >= SMDBLK)
248  {
249  const size_t ALNLEN = Length - (Length % SMDBLK);
250 
251  while (pctr != ALNLEN)
252  {
253 #if defined(__AVX512__)
254  AND512(Input, InOffset + (pctr / INPLEN), Output, OutOffset + (pctr / OTPLEN));
255 #elif defined(__AVX2__)
256  AND256(Input, InOffset + (pctr / INPLEN), Output, OutOffset + (pctr / OTPLEN));
257 #elif defined(__AVX__)
258  AND128(Input, InOffset + (pctr / INPLEN), Output, OutOffset + (pctr / OTPLEN));
259 #endif
260  pctr += SMDBLK;
261  }
262  }
263 #endif
264  for (i = 0; i < Length - (pctr * ELMLEN); ++i)
265  {
266  Output[OutOffset + pctr] &= Input[InOffset + pctr];
267  }
268  }
269  }

280  inline static void AND128(const ArrayA &Input, size_t InOffset, ArrayB &Output, size_t OutOffset)
281  {
282  const size_t INPLEN = sizeof(ArrayA::value_type);
283  const size_t OTPLEN = sizeof(ArrayB::value_type);
284 
285  CEXASSERT((Input.size() - InOffset) * INPLEN >= 16, "Length is larger than input size");
286  CEXASSERT((Output.size() - OutOffset) * OTPLEN >= 16, "Length is larger than output size");
287 
288 #if defined(__AVX__)
289  _mm_storeu_si128(reinterpret_cast<__m128i*>(&Output[OutOffset]), _mm_and_si128(_mm_loadu_si128(reinterpret_cast<const __m128i*>(&Input[InOffset])), _mm_loadu_si128(reinterpret_cast<__m128i*>(&Output[OutOffset]))));
290 #else
291  for (size_t i = 0; i < (16 / OTPLEN); ++i)
292  {
293  Output[OutOffset + i] &= Input[InOffset + i];
294  }
295 #endif
296  }

411  template<typename Array>
412  inline static bool Compare(const Array &A, size_t AOffset, const Array &B, size_t BOffset, size_t Elements)
413  {
414  size_t diff;
415 
416  diff = 0;
417 
418  for (size_t i = 0; i != Elements; ++i)
419  {
420  diff |= (A[AOffset + i] ^ B[BOffset + i]);
421  }
422 
423  return diff == 0;
424  }
