92  template <typename Array>
93  inline static void Clear(Array &Output, size_t Offset, size_t Length)
94  {
95  if (Length != 0)
96  {
97  const size_t ELMLEN = sizeof(Array::value_type);
98  CEXASSERT((Output.size() - Offset) * ELMLEN >= Length, "Length is larger than output capacity");
99  CEXASSERT(ELMLEN <= Length, "Integer type is larger than length");
100 
101  size_t prcCtr = 0;
102 
103 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
104 # if defined(__AVX512__)
105  const size_t SMDBLK = 64 / ELMLEN;
106 # elif defined(__AVX2__)
107  const size_t SMDBLK = 32 / ELMLEN;
108 # else
109  const size_t SMDBLK = 16 / ELMLEN;
110 # endif
111 
112  if (Length / ELMLEN >= SMDBLK)
113  {
114  const size_t ALNLEN = (Length / (SMDBLK * ELMLEN)) * SMDBLK;
115 
116  while (prcCtr != ALNLEN)
117  {
118 #if defined(__AVX512__)
119  CLEAR512(Output, Offset + prcCtr);
120 #elif defined(__AVX2__)
121  CLEAR256(Output, Offset + prcCtr);
122 #elif defined(__AVX__)
123  CLEAR128(Output, Offset + prcCtr);
124 #endif
125  prcCtr += SMDBLK;
126  }
127  }
128 #endif
129 
130  if (prcCtr * ELMLEN != Length)
131  {
132  std::memset(&Output[Offset + prcCtr], 0, Length - (prcCtr * ELMLEN));
133  }
134  }
135  }

203  template<typename Array>
204  inline static bool Compare(const Array &A, size_t AOffset, const Array &B, size_t BOffset, size_t Elements)
205  {
206  CEXASSERT((A.size() - AOffset) >= Length, "Length is larger than A size");
207  CEXASSERT((B.size() - BOffset) >= Length, "Length is larger than B size");
208 
209  size_t diff = 0;
210 
211  for (size_t i = 0; i != Elements; ++i)
212  {
213  diff |= (A[AOffset + i] ^ B[BOffset + i]);
214  }
215 
216  return diff == 0;
217  }
