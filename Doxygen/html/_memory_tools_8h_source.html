89  template <typename Array>
90  inline static void Clear(Array &Output, size_t Offset, size_t Length)
91  {
92  if (Length != 0)
93  {
94  const size_t ELMLEN = sizeof(Array::value_type);
95  CEXASSERT((Output.size() - Offset) * ELMLEN >= Length, "Length is larger than output capacity");
96  CEXASSERT(ELMLEN <= Length, "Integer type is larger than length");
97 
98  size_t prcCtr = 0;
99 
100 #if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512__)
101 # if defined(__AVX512__)
102  const size_t SMDBLK = 64 / ELMLEN;
103 # elif defined(__AVX2__)
104  const size_t SMDBLK = 32 / ELMLEN;
105 # else
106  const size_t SMDBLK = 16 / ELMLEN;
107 # endif
108 
109  if (Length / ELMLEN >= SMDBLK)
110  {
111  const size_t ALNLEN = (Length / (SMDBLK * ELMLEN)) * SMDBLK;
112 
113  while (prcCtr != ALNLEN)
114  {
115 #if defined(__AVX512__)
116  CLEAR512(Output, Offset + prcCtr);
117 #elif defined(__AVX2__)
118  CLEAR256(Output, Offset + prcCtr);
119 #elif defined(__AVX__)
120  CLEAR128(Output, Offset + prcCtr);
121 #endif
122  prcCtr += SMDBLK;
123  }
124  }
125 #endif
126 
127  if (prcCtr * ELMLEN != Length)
128  {
129  std::memset(&Output[Offset + prcCtr], 0, Length - (prcCtr * ELMLEN));
130  }
131  }
132  }

197  template<typename Array>
198  inline static bool Compare(const Array &A, size_t AOffset, const Array &B, size_t BOffset, size_t Elements)
199  {
200  CEXASSERT((A.size() - AOffset) >= Length, "Length is larger than A size");
201  CEXASSERT((B.size() - BOffset) >= Length, "Length is larger than B size");
202 
203  size_t diff = 0;
204 
205  for (size_t i = 0; i != Elements; ++i)
206  {
207  diff |= (A[AOffset + i] ^ B[BOffset + i]);
208  }
209 
210  return diff == 0;
211  }
